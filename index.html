// Category names const categoryNames = { ai: 'AI & ML', frontend: '前端', backend: '后端', devops: 'DevOps', tools: '工具', learning: '学习', others: '其他' }; // Initialize document.addEventListener('DOMContentLoaded', () => { updateStats(); renderRepos(); }); // Update Stats function updateStats() { document.getElementById('header-total').textContent = repos.length; document.getElementById('header-starred').textContent = repos.filter(r => r.starred).length; } // Render Repos function renderRepos() { const list = document.getElementById('repo-list'); const emptyState = document.getElementById('empty-state'); let filtered = repos; // Filter by category if (currentCategory !== 'all') { filtered = filtered.filter(r => r.category === currentCategory); } // Filter by search if (searchQuery) { const q = searchQuery.toLowerCase(); filtered = filtered.filter(r => r.name.toLowerCase().includes(q) || r.owner.toLowerCase().includes(q) || r.title?.toLowerCase().includes(q) || r.notes?.toLowerCase().includes(q) ); } // Sort by starred first, then by date filtered.sort((a, b) => { if (a.starred !== b.starred) return b.starred ? 1 : -1; return new Date(b.addedAt) - new Date(a.addedAt); }); if (filtered.length === 0) { list.innerHTML = ''; emptyState.classList.remove('hidden'); return; } emptyState.classList.add('hidden'); list.innerHTML = filtered.map(repo => `
${repo.starred ? '
' : ''}
${categoryNames[repo.category] || '其他'}

${repo.owner}/${repo.name}

${repo.title ? `

${repo.title}

` : ''} ${repo.notes ? `

${repo.notes}

` : repo.description ? `

${repo.description}

` : ''}
${repo.language ? ` ${repo.language}` : ''} ${repo.stars ? ` ${repo.stars}` : ''}
`).join(''); } // Filter by Category function filterByCategory(category) { currentCategory = category; document.querySelectorAll('.category-tag').forEach(btn => { if (btn.dataset.category === category) { btn.classList.remove('bg-white', 'text-gray-600', 'border', 'border-gray-200'); btn.classList.add('bg-primary', 'text-white', 'shadow-md'); } else { btn.classList.remove('bg-primary', 'text-white', 'shadow-md'); btn.classList.add('bg-white', 'text-gray-600', 'border', 'border-gray-200'); } }); renderRepos(); } // Search function searchRepos() { searchQuery = document.getElementById('search-input').value.trim(); renderRepos(); } // Toggle Star function toggleStar(id) { const repo = repos.find(r => r.id === id); if (repo) { repo.starred = !repo.starred; saveRepos(); renderRepos(); updateStats(); showToast(repo.starred ? '已标记为精选' : '已取消精选'); } } // Delete Repo function deleteRepo(id) { if (confirm('确定要删除这个仓库吗？')) { repos = repos.filter(r => r.id !== id); saveRepos(); renderRepos(); updateStats(); showToast('已删除'); } } // Fetch Repo Info async function fetchRepoInfo() { const url = document.getElementById('repo-url').value.trim(); if (!url) return; const match = url.match(/github\.com\/([^\/]+)\/([^\/]+)/); if (!match) { showToast('无效的 GitHub URL'); return; } const [, owner, name] = match; // 显示预览 document.getElementById('repo-preview').classList.remove('hidden'); document.getElementById('preview-avatar').src = `https://github.com/${owner}.png`; document.getElementById('preview-name').textContent = `${owner}/${name.replace(/\.git$/, '')}`; document.getElementById('preview-desc').textContent = '正在获取仓库信息...'; document.getElementById('preview-stars').textContent = '-'; document.getElementById('preview-lang').textContent = '-'; // 尝试获取 GitHub API 数据 try { const response = await fetch(`https://api.github.com/repos/${owner}/${name.replace(/\.git$/, '')}`); if (response.ok) { const data = await response.json(); document.getElementById('preview-desc').textContent = data.description || '暂无描述'; document.getElementById('preview-stars').textContent = (data.stargazers_count >= 1000) ? (data.stargazers_count / 1000).toFixed(1) + 'k' : data.stargazers_count; document.getElementById('preview-lang').textContent = data.language || 'Unknown'; } else { document.getElementById('preview-desc').textContent = '无法获取仓库信息'; } } catch (error) { document.getElementById('preview-desc').textContent = '无法获取仓库信息'; } } // Add Repo function addRepo() { const url = document.getElementById('repo-url').value.trim(); const category = document.querySelector('input[name="category"]:checked')?.value; const title = document.getElementById('repo-title').value.trim(); const notes = document.getElementById('repo-notes').value.trim(); const starred = document.getElementById('repo-starred').checked; if (!url) { showToast('请输入 GitHub URL'); return; } if (!category) { showToast('请选择分类'); return; } const match = url.match(/github\.com\/([^\/]+)\/([^\/]+)/); if (!match) { showToast('无效的 GitHub URL'); return; } const [, owner, name] = match; // Check if exists const exists = repos.find(r => r.url === url); if (exists) { showToast('该仓库已存在'); return; } const repo = { id: Date.now().toString(), url, owner, name: name.replace(/\.git$/, ''), description: document.getElementById('preview-desc').textContent || '', language: document.getElementById('preview-lang').textContent || '', stars: document.getElementById('preview-stars').textContent || '', category, title, notes, starred, addedAt: new Date().toISOString() }; repos.unshift(repo); saveRepos(); hideAddModal(); resetForm(); renderRepos(); updateStats(); showToast('仓库已添加'); } // Reset Form function resetForm() { document.getElementById('repo-url').value = ''; document.querySelectorAll('input[name="category"]').forEach(r => r.checked = false); document.getElementById('repo-title').value = ''; document.getElementById('repo-notes').value = ''; document.getElementById('repo-starred').checked = false; document.getElementById('repo-preview').classList.add('hidden'); } // Save to LocalStorage function saveRepos() { localStorage.setItem('ezgit-repos', JSON.stringify(repos)); } // Modal Functions function showAddModal() { document.getElementById('add-modal').classList.remove('hidden'); } function hideAddModal() { document.getElementById('add-modal').classList.add('hidden'); resetForm(); } // Toast function showToast(message) { const toast = document.getElementById('toast'); document.getElementById('toast-message').textContent = message; toast.classList.remove('hidden'); setTimeout(() => toast.classList.add('hidden'), 3000); } // Close modal on outside click document.getElementById('add-modal').addEventListener('click', (e) => { if (e.target.id === 'add-modal') hideAddModal(); }); document.addEventListener('keydown', (e) => { if (e.key === 'Escape') hideAddModal(); });
